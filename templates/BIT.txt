
// BIT
struct BIT {
    int n;
    vi bit;

    BIT(int n): n(n), bit(n+1, 0) {
    }

    int sum(int i) {
        int s = 0;
        while (i > 0) {
            s += bit[i];
            i -= i & -i;
        }
        return s;
    }

    void add(int i, int x) {
        while (i <= n) {
            bit[i] += x;
            i += i & -i;
        }
    }
};
